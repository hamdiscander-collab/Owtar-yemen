export default async function handler(req, res) { if (req.method !== 'POST') return res.status(405).send('Method Not Allowed'); const { prompt } = req.body; const HF_TOKEN = process.env.HF_TOKEN; try { const response = await fetch( "https://api-inference.huggingface.co/models/facebook/musicgen-small", { headers: { Authorization: `Bearer ${HF_TOKEN}` }, method: "POST", body: JSON.stringify({ inputs: "Yemeni Oud music, " + prompt }), } ); // إذا كان السيرفر نائماً (503) if (response.status === 503) { return res.status(503).json({ error: "السيرفر يستيقظ، يرجى المحاولة بعد 15 ثانية" }); } if (!response.ok) throw new Error("فشل الاتصال بالسيرفر"); const arrayBuffer = await response.arrayBuffer(); res.setHeader('Content-Type', 'audio/mpeg'); res.send(Buffer.from(arrayBuffer)); } catch (error) { res.status(500).json({ error: "خطأ فني في السيرفر" }); } }
