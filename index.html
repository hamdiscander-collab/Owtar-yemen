function startAIGeneration() { const text = document.getElementById('promptInput').value; if(!text) { alert("يرجى كتابة نص أو فكرة أولاً"); return; } const log = document.getElementById('log'); log.innerHTML = ""; // تنظيف السجل addLog("جاري تحليل الكلمات: " + text.substring(0, 15) + "..."); setTimeout(() => addLog("جاري ضبط مقام العود اليمني... 🎸"), 1500); setTimeout(() => { addLog("تم الإنتاج! يمكنك الاستماع الآن ✅"); // هنا قمت بتحديث الرابط لمقطع صوتي حقيقي (كمثال) const player = document.getElementById('audioPlayback'); // ملاحظة: هذا رابط تجريبي لمقطع موسيقي، يمكنك استبداله برابط ملف MP3 يمني player.src = "https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3"; player.style.display = "block"; player.play(); // تشغيل تلقائي بعد المعالجة }, 4000); }
